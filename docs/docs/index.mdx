# Welcome to RushDB

RushDB is an open-source database built as a self-aware memory layer for agents, humans, and apps. Push any JSON — RushDB infers types, links nested objects into a graph, and makes everything queryable by field value or by meaning.

No schema design. No migrations. No separate vector store.

---

## How it works

<ResponsiveTabs>
  <TabItem value="agent" label="Agent memory" default>

An agent stores observations and retrieves them later by meaning:

<LanguageTabs>
<TabItem value="python" label="Python" default>

```python
from rushdb import RushDB
db = RushDB("RUSHDB_API_KEY")

# Agent pushes structured observations
db.records.create_many(
    label="MEMORY",
    data=[
        {"content": "User prefers concise answers with code examples", "tags": ["preference"]},
        {"content": "Project deadline is end of Q2", "tags": ["deadline"]},
        {"content": "Run evals before merging to main", "tags": ["workflow"]},
    ]
)

# Later: retrieve by meaning, not exact keyword
response = db.ai.search({
    "propertyName": "content",
    "query": "how should I format my responses",
    "labels": ["MEMORY"],
    "limit": 3
})
# [0.94] User prefers concise answers with code examples
```

An app stores articles and queries them by tag or by topic:

<LanguageTabs>
<TabItem value="python" label="Python" default>

```python
from rushdb import RushDB
db = RushDB("RUSHDB_API_KEY")

# Push a batch of articles
db.records.create_many(
    label="ARTICLE",
    data=[
        {"title": "Intro to Neo4j",     "content": "Graph databases store...", "tags": ["databases"]},
        {"title": "Vector Search 101",  "content": "Embeddings represent...",  "tags": ["ai", "search"]},
        {"title": "ACID Transactions",  "content": "Atomicity ensures...",     "tags": ["databases"]},
    ]
)

# Structured query — find by tag
db_articles = db.records.find({
    "labels": ["ARTICLE"],
    "where": {"tags": {"$in": ["databases"]}}
})

# Semantic query — find by meaning, pre-filtered by tag
results = db.ai.search({
    "propertyName": "content",
    "query": "how graphs store relationships",
    "labels": ["ARTICLE"],
    "where": {"tags": {"$in": ["databases"]}}
})
```

Product telemetry arrives with full nested context — RushDB builds the graph ontology on write:

<LanguageTabs>
<TabItem value="python" label="Python" default>

```python
from rushdb import RushDB
db = RushDB("RUSHDB_API_KEY")

# Ingest raw sessions — nested objects become linked records automatically
db.records.create_many(
    label="SESSION",
    data={
        "startedAt": "2026-03-21T09:14:00Z",
        "user": {"plan": "pro", "region": "eu-west"},
        "events": [
            {"type": "feature_use", "feature": "semantic_search", "durationMs": 450, "ok": True},
            {"type": "feature_use", "feature": "bulk_import",     "durationMs": 80,  "ok": False},
        ]
    }
)
# RushDB created: SESSION → USER, SESSION → EVENT × 2
# No schema defined — ontology emerged from the data shape

# Structured: find slow, successful operations
slow = db.records.find({
    "labels": ["EVENTS"],
    "where": {"ok": True, "durationMs": {"$gte": 300}}
})

# Semantic: agent asks a freeform analytics question
candidates = db.ai.search({
    "propertyName": "feature",
    "query": "operations related to uploading or importing data",
    "labels": ["EVENT"],
    "where": {"ok": False},
    "limit": 10
})
# → bulk_import, csv_upload, api_push — discovered without knowing field values upfront
```

Build a small graph manually, then explore all 5 ontology perspectives — labels, properties, values, relationships, and records:

<LanguageTabs>
<TabItem value="python" label="Python" default>

```python
from rushdb import RushDB
db = RushDB("RUSHDB_API_KEY")

# ── Create records ────────────────────────────────────────────────────────
movies = db.records.create_many(
    label="MOVIE",
    data=[
        {"title": "Inception",        "genre": "sci-fi", "rating": 8.8},
        {"title": "The Aviator",      "genre": "drama",  "rating": 7.5},
        {"title": "The Last Samurai", "genre": "drama",  "rating": 7.7},
    ],
    options={"returnResult": True}
)
inception, aviator, samurai = movies

actors = db.records.create_many(
    label="ACTOR",
    data=[
        {"name": "Leonardo DiCaprio", "country": "USA"       },
        {"name": "Cate Blanchett",    "country": "Australia" },
        {"name": "Ken Watanabe",      "country": "Japan"     },
    ],
    options={"returnResult": True}
)
leo, cate, ken = actors

directors = db.records.create_many(
    label="DIRECTOR",
    data=[
        {"name": "Christopher Nolan", "country": "UK"},
        {"name": "Ridley Scott",      "country": "UK"},
    ],
    options={"returnResult": True}
)
nolan, scott = directors

# ── Wire up relationships ─────────────────────────────────────────────────
db.records.attach(source=leo,   target=[inception, aviator], options={"type": "ACTED_IN"})
db.records.attach(source=cate,  target=aviator,              options={"type": "ACTED_IN"})
db.records.attach(source=ken,   target=[inception, samurai], options={"type": "ACTED_IN"})
db.records.attach(source=nolan, target=inception,            options={"type": "DIRECTED"})
db.records.attach(source=scott, target=samurai,              options={"type": "DIRECTED"})

# ── Labels ────────────────────────────────────────────────────────────────
# Which labels exist in the project?
label_counts = db.labels.find()
# → ['MOVIE', 'ACTOR', 'DIRECTOR']

# Only labels that have at least one record with rating ≥ 8
high_rated_labels = db.labels.find({"where": {"rating": {"$gte": 8}}})
# → ['MOVIE']

# ── Properties ────────────────────────────────────────────────────────────
# What fields exist on MOVIE records, and what are their inferred types?
movie_props = db.properties.find({"labels": ["MOVIE"]})
# → [{"name": "title", "type": "string"}, {"name": "genre", "type": "string"}, {"name": "rating", "type": "number"}]

# ── Values ────────────────────────────────────────────────────────────────
# What distinct values exist for 'rating', and what is the range?
rating_prop = next(p for p in movie_props if p["name"] == "rating")
rating_range = db.properties.values(property_id=rating_prop["id"])
# → {"values": [7.5, 7.7, 8.8], "min": 7.5, "max": 8.8}

# All distinct genres in the dataset
genre_prop = next(p for p in movie_props if p["name"] == "genre")
genre_list = db.properties.values(property_id=genre_prop["id"])
# → {"values": ["drama", "sci-fi"]}

# ── Relationships ─────────────────────────────────────────────────────────
# What edges exist in the graph?
all_rels = db.relationships.find()
# → [{"sourceLabel": "ACTOR", "targetLabel": "MOVIE", "type": "ACTED_IN"},
#    {"sourceLabel": "DIRECTOR", "targetLabel": "MOVIE", "type": "DIRECTED"}, ...]

# ── Records — same db.records.find(), different entry points ──────────────

# 1. Actors in sci-fi films
db.records.find({"labels": ["ACTOR"],
    "where": {"MOVIE": {"genre": "sci-fi"}}
})
# → Leonardo DiCaprio, Ken Watanabe

# 2. Films with an Australian actor
db.records.find({"labels": ["MOVIE"],
    "where": {"ACTOR": {"country": "Australia"}}
})
# → The Aviator

# 3. Directors of films where a Japanese actor appeared  (multi-hop)
db.records.find({"labels": ["DIRECTOR"],
    "where": {"MOVIE": {"ACTOR": {"country": "Japan"}}}
})
# → Christopher Nolan, Ridley Scott

# 4. UK-directed films rated above 8
db.records.find({"labels": ["MOVIE"],
    "where": {"DIRECTOR": {"country": "UK"}, "rating": {"$gte": 8}}
})
# → Inception

# 5. Actors in films directed by Christopher Nolan  (multi-hop)
db.records.find({"labels": ["ACTOR"],
    "where": {"MOVIE": {"DIRECTOR": {"name": "Christopher Nolan"}}}
})
# → Leonardo DiCaprio, Ken Watanabe
```

## What RushDB does

**Ingest anything.** Push a flat object, a nested tree, or a batch — RushDB walks the structure, infers types, and stores each level as a linked record. No upfront schema.

**Query by value.** Filter by any field with operators like `$gte`, `$in`, `$contains`, and `$startsWith`. Traverse relationships directly inside the `where` clause.

**Query by meaning.** Create an embedding index on any text property. `db.ai.search()` returns records ranked by semantic similarity, with an optional `where` prefilter to narrow the candidate set first.

**Stay consistent.** Full ACID transactions wrap any combination of writes, relationship changes, and reads.
